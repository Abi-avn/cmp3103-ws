#!/usr/bin/env python3

import rclpy
from rclpy.node import Node
from sensor_msgs.msg import Image, LaserScan
from geometry_msgs.msg import Twist
from cv_bridge import CvBridge
import cv2
import numpy as np


class TidyBotLevel1(Node):

    def __init__(self):
        super().__init__('tidybot_level1')

        self.cmd_pub = self.create_publisher(Twist, 'cmd_vel', 10)
        self.create_subscription(Image, '/limo/depth_camera_link/image_raw', self.image_callback, 10)
        self.create_subscription(LaserScan, '/scan', self.scan_callback, 10)

        self.bridge = CvBridge()

        self.front_distance = float('inf')
        self.state = "SEARCH"

        self.PATCH_THRESHOLD = 30000

        # memory for lost cube tracking
        self.last_seen_cube = None
        self.lost_counter = 0

        print("🚀 FINAL TIDYBOT LEVEL 1 STARTED")

    # -------------------------
    # LIDAR
    # -------------------------
    def scan_callback(self, msg):
        ranges = msg.ranges
        front = ranges[len(ranges)//2 - 10 : len(ranges)//2 + 10]
        front = [r for r in front if r > 0.0 and not np.isinf(r)]
        self.front_distance = min(front) if front else float('inf')

    # -------------------------
    # CAMERA
    # -------------------------
    def image_callback(self, msg):

        frame = self.bridge.imgmsg_to_cv2(msg, desired_encoding='bgr8')
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        h, w = frame.shape[:2]

        # -------------------------
        # RED PATCH
        # -------------------------
        lower_red1 = np.array([0, 70, 50])
        upper_red1 = np.array([10, 255, 255])
        lower_red2 = np.array([170, 70, 50])
        upper_red2 = np.array([180, 255, 255])

        mask_red = cv2.inRange(hsv, lower_red1, upper_red1) + \
                   cv2.inRange(hsv, lower_red2, upper_red2)

        red_area = np.sum(mask_red)
        red_visible = red_area > self.PATCH_THRESHOLD

        # -------------------------
        # BROWN CUBES
        # -------------------------
        lower_brown = np.array([5, 50, 50])
        upper_brown = np.array([25, 255, 255])

        mask_brown = cv2.inRange(hsv, lower_brown, upper_brown)

        contours, _ = cv2.findContours(mask_brown, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)

        cubes_on_red = []

        for cnt in contours:
            area = cv2.contourArea(cnt)
            if area < 150:
                continue

            M = cv2.moments(cnt)
            if M['m00'] == 0:
                continue

            cx = int(M['m10'] / M['m00'])
            cy = int(M['m01'] / M['m00'])

            cube_mask = np.zeros_like(mask_red)
            cv2.drawContours(cube_mask, [cnt], -1, 255, -1)

            overlap = np.sum(cv2.bitwise_and(mask_red, cube_mask))

            # allow full + partial patch coverage
            if overlap > 50 or (red_visible and cy > h * 0.6):
                cubes_on_red.append((cnt, cx, cy, area))

        twist = Twist()

        # =========================
        # STATE MACHINE
        # =========================

        # -------------------------
        # SEARCH STATE
        # -------------------------
        if self.state == "SEARCH":

            twist.angular.z = 0.3
            twist.linear.x = 0.05

            if red_visible and cubes_on_red:
                print("🎯 Target acquired")
                self.state = "TARGET"

        # -------------------------
        # TARGET STATE
        # -------------------------
        elif self.state == "TARGET":

            if cubes_on_red:
                cubes_on_red.sort(key=lambda x: abs(x[1] - w // 2))
                _, cx, _, _ = cubes_on_red[0]

                self.last_seen_cube = cx
                self.lost_counter = 0

            else:
                self.lost_counter += 1

                if self.lost_counter < 5 and self.last_seen_cube is not None:
                    cx = self.last_seen_cube
                else:
                    if red_visible:
                        print("❌ Lost cube → back to SEARCH")
                        self.state = "SEARCH"
                    else:
                        print("❌ Lost cube → SPIN_FOR_TARGET")
                        self.state = "SPIN_FOR_TARGET"
                    return

            error = cx - w // 2
            twist.angular.z = -0.003 * error
            twist.linear.x = 0.15

            if abs(error) < 30:
                print("🚀 START PUSH")
                self.state = "PUSH"

        # -------------------------
        # PUSH STATE
        # -------------------------
        elif self.state == "PUSH":

            twist.linear.x = 0.4
            twist.angular.z = 0.0

            if self.front_distance < 0.3:
                print("🧱 Wall reached → TURN")
                self.state = "TURN"

        # -------------------------
        # TURN STATE
        # -------------------------
        elif self.state == "TURN":

            twist.angular.z = 0.6
            twist.linear.x = 0.0

            if self.front_distance > 0.8:
                print("🔄 Sweep mode")
                self.state = "SWEEP"

        # -------------------------
        # SWEEP STATE
        # -------------------------
        elif self.state == "SWEEP":

            twist.angular.z = 0.4

            if red_visible:
                print("🔁 Back to SEARCH")
                self.state = "SEARCH"

        # -------------------------
        # SPIN FOR TARGET (NEW FIX)
        # -------------------------
        elif self.state == "SPIN_FOR_TARGET":

            twist.angular.z = 0.5
            twist.linear.x = 0.0

            if red_visible and cubes_on_red:
                print("🎯 Cube reacquired → TARGET")
                self.state = "TARGET"

        # -------------------------
        # OBSTACLE SAFETY
        # -------------------------
        if self.front_distance < 0.3:
            twist.linear.x = 0.0
            twist.angular.z = 0.6

        self.cmd_pub.publish(twist)

        # -------------------------
        # DEBUG
        # -------------------------
        cv2.putText(frame, f"STATE: {self.state}",
                    (10, 25), cv2.FONT_HERSHEY_SIMPLEX, 0.7, (255,255,255), 2)

        cv2.imshow("Camera", cv2.resize(frame, (0, 0), fx=0.4, fy=0.4))
        cv2.imshow("Red Mask", cv2.resize(mask_red, (0, 0), fx=0.4, fy=0.4))
        cv2.imshow("Brown Mask", cv2.resize(mask_brown, (0, 0), fx=0.4, fy=0.4))
        cv2.waitKey(1)


def main():
    rclpy.init()
    node = TidyBotLevel1()
    rclpy.spin(node)
    node.destroy_node()
    rclpy.shutdown()
    cv2.destroyAllWindows()


if __name__ == '__main__':
    main()
